## TextEllipsis

**TextEllipsis** is an automatic text length recognition system that displays text beyond the parent element, omits the original content of the text in the tooltip, and does not exceed the displayed text content.

<Tabs items={['component', 'code']}>
  <Tabs.Tab>
    <TextEllipsisDemo isExceed />
    <br/>
    <TextEllipsisDemo isExceed={false} />
  </Tabs.Tab>
  <Tabs.Tab>
    ```jsx {2, 12-16} filename="TextEllipsisDemo.tsx"
    import React from "react"
    import { TextEllipsis, cn } from 'nk-plus-antd'

    const TextEllipsisDemo: React.FC = ({ isExceed }: { isExceed: boolean }) => {
      return (
        <div 
          className={cn(
            "h-[24px] border border-gray-100 truncate",
            isExceed ? "w-[250px]" : "w-[520px]"
          )}
        >
          <TextEllipsis 
            text={'This is a component with text exceeding the displayed tooltip.'}
            hover
            placement={'top'}
          />
        </div>
      )
    }

    export default TextEllipsisDemo
    ```
  </Tabs.Tab>
</Tabs>     

## CodeBrowse

**CodeBrowse**  is a beautiful code monitor that can copy code content.

<Tabs items={['code']}>
  <Tabs.Tab>
    ```jsx {2, 23-25} filename="CodeBrowseDemo.tsx"
      import React from "react"
      import { CodeBrowse } from 'nk-plus-antd'
      import { message } from 'antd'

      const code = `
        import React from 'react'
        import { AntdRegistry } from '@ant-design/nextjs-registry'

        const RootLayout = ({ children }: React.PropsWithChildren) => (
          <html lang="en">
            <body>
              <AntdRegistry>{children}</AntdRegistry>
            </body>
          </html>
        )

        export default RootLayout
      `

      const CodeBrowseDemo: React.FC = () => {
        return (
          <div>
            <CodeBrowse code={code} title={'bower.tsx'} callCopy={(bool) => {
              if(bool) message.success('复制成功😄')
            }} />
          </div>
        )
      }

      export default CodeBrowseDemo
    ```
  </Tabs.Tab>
